Напишите функцию простейшего калькулятора. Пользователь вводит число a, число b и вид арифметической 
операции над ними. Функция возвращает результат вычисления.



def arithmetic (a, b, c):                                   # Функция простейшего калькулятора
    if c == "+":                                            # Если знак "+"
        return a + b                                        # Результат - сумма чисел
    elif c == "-":                                          # Если знак "-"
        return a - b                                        # Результат - разность чисел
    elif c == "*":                                          # Если знак "*"
        return a * b                                        # Результат - произведение чисел
    elif c == "/":                                          # Если знак "/"
        return a / b                                        # Результат - частное чисел
    else:                                                   # В противном случае
        return "Неизвестная операция!"                      # Сообщение об ошибке
result = arithmetic (int(input("Введите число a: ")),         # Вызов функции
                     int(input("Введите число b: ")), 
                     input("Введите операцию: "))
print("Результат: ", result)                                # Вывод результата
